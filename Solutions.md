6- Find Product using modulo 1000000007
#include <stdio.h>

#define MOD 1000000007

int main() {
    int N;
    long long int product = 1;
    scanf("%d", &N);
    int arr[N];
    for (int i = 0; i < N; i++) {
        scanf("%d", &arr[i]);
    }
    for (int i = 0; i < N; i++) {
        product = (product * arr[i]) % MOD;
    }
    printf("%lld\n", product);
    return 0;
}

7- String Palindrome
#include <stdio.h>
#include <string.h>

int isPalindrome(char S[]) {
    int len = strlen(S);
    int i, j;
    for (i = 0, j = len - 1; i < j; i++, j--) {
        if (S[i] != S[j]) {
            return 0; 
        }
    }
    return 1;  
}
int main() {
    char S[100];
    scanf("%s", S);
    if (isPalindrome(S)) {
        printf("YES\n");
    } else {
        printf("NO\n");
    }
    return 0;
}
